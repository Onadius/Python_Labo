
$ tleap -s -f leaprc.ff99SB //作業ディレクトリ内で

> set default PBradii mbondi2

> test = sequence{ACE ASH NME} //testは変数にあたる

> impose test {2} {{OE1 CG OE2 HE2 -90 }} //-180 ~ 180でシミュレーション

> saveamberparm test confamb.prmtop confamb.inpcrd //360この構造ファイル(.prmtop, .impcrdを)保存

> quit
